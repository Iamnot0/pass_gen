// üîê Generate a secure passphrase with 2 unique words + random number
function generatePassphrase(wordList = wordList) {
  if (wordList.length < 2) {
    throw new Error("The wordList must contain at least 3 words.");
  }
  
  let selectedWords = [];
  while (selectedWords.length < 2) {
    let randomIndex = Math.floor(Math.random() * wordList.length);
    let word = wordList[randomIndex];
    if (!selectedWords.includes(word)) {
      selectedWords.push(word);
    }
  }

  selectedWords = selectedWords.map(word =>
    word.charAt(0).toUpperCase() + word.slice(1)
  );

  let insertPosition = Math.floor(Math.random() * selectedWords.length);
  let randomNumber = Math.floor(Math.random() * 10);

  selectedWords.splice(insertPosition + 1, 0, randomNumber.toString());

  return selectedWords.join('');
}

// üì• Insert passphrase into currently active cell
function enterTextInCurrentCell(text) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  const cell = sheet.getActiveCell();
  cell.setValue(text);
}

// üîÅ Replace all cells in Column J labeled "password" with generated passphrases
function replacePasswords() {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet();
  const column = 10; // Column J
  const startRow = 2; // Skip header
  const lastRow = sheet.getLastRow();

  for (let row = startRow; row <= lastRow; row++) {
    const cell = sheet.getRange(row, column);
    const cellValue = cell.getValue();

    if (typeof cellValue === 'string' && cellValue.toLowerCase() === "password") {
      const passphrase = generatePassphrase(wordList);
      cell.setValue(passphrase);
      Utilities.sleep(300); // optional delay
    }
  }
}
